   120→@pytest.mark.asyncio
   121→@pytest.mark.parametrize(["plan_bundled_cpus", "plan_bundled_mem_gib"], [[None, None], [None, 1], [2, None], [3, None]])
   122→@pytest.mark.parametrize(["sandbox", "nonpreemptible"], [[True, False], [False, True], [False, False]])
   123→async def test_track_task_metrics(
   124→    state: State, user, clear_metrics_tables, plan_bundled_cpus, plan_bundled_mem_gib, sandbox, nonpreemptible
   125→):
   126→    """Tests that metrics are produced and consumed correctly."""
   127→    # generate messages using the timeseries manager interface
   128→    message_value = "test"
   129→    message = _generate_message(message_value, 0)
   130→    metrics = [
   131→        internal_api_pb2.ResourceMetric(resource="elapsed_ns", value=1),
   132→        internal_api_pb2.ResourceMetric(resource="cpu_ns", value=3),
   133→        internal_api_pb2.ResourceMetric(resource="mem_ns", value=4),
   134→        internal_api_pb2.ResourceMetric(resource="gpu_ns", value=1),
   135→        internal_api_pb2.ResourceMetric(resource="gpu_mem_ns", value=1),
   136→        internal_api_pb2.ResourceMetric(resource="gpu_temp_ns", value=1),
   137→    ]
   138→    resources = api_pb2.Resources(
   139→        milli_cpu=500,  # 500 milli-CPU = 0.5 CPU core
   140→        memory_mb=2048,  # 2048 MiB = 2 GiB
   141→        gpu_config=api_pb2.GPUConfig(count=1, type=api_pb2.GPU_TYPE_H100),
   142→    )
   143→    message.pop("timestamp")
   144→
   145→    workspace_id = user.workspace_id
   146→    workspace_struct = await state.durable.workspaces[workspace_id]
   147→
   148→    existing_ephemeral_plan_struct = await state.ephemeral.plans[workspace_struct.plan_id]
   149→    new_values = {}
   150→    if plan_bundled_cpus is not None:
   151→        new_values["bundled_cpus"] = plan_bundled_cpus
   152→    if plan_bundled_mem_gib is not None:
   153→        new_values["bundled_mem_gib"] = plan_bundled_mem_gib
   154→    durable_plan_struct = await copy_plan(state, existing_ephemeral_plan_struct.key, **new_values)
   155→    ephemeral_plan_struct = await state.ephemeral.plans[durable_plan_struct.key]
   156→
   157→    # Create a dummy SchedulerPlacement protobuf message to construct regions list
   158→    temp_message = api_pb2.SchedulerPlacement()
   159→    regions = temp_message.regions
   160→    regions.append("us-east-1")
   161→    regions.extend(["us-west-2", "eu-west-1"])
   162→
   163→    await state.task_metrics_store.track_task_metrics(
   164→        task_id=message["task_id"],
   165→        workspace_id="ac-test",
   166→        environment_id="en-test",
   167→        billing_type=message["billing_type"],
   168→        app_id="ap-test",
   169→        app_type="deployed",
   170→        metrics=metrics,
   171→        resources=resources,
   172→        ephemeral_plan_struct=ephemeral_plan_struct,
   173→        regions=regions,
   174→        function_id=message["function_id"] if not sandbox else None,
   175→        function_name="name-of-" + message["function_id"],
   176→        image_id=message["image_id"],
   177→        sandbox_id=message["sandbox_id"] if sandbox else None,
   178→        nonpreemptible=nonpreemptible,
   179→        base_function_id=None,
   180→    )
   181→    await state.mock_clickpipes.task_metrics.join()
   182→
   183→    # verify that messages have been ingested & message content
   184→    query_result = await state.task_metrics_store.ch_client.query(AllTaskMetricsQuery())
   185→    rows = query_result.named_results()
   186→    row = next(rows)
   187→    assert row["workspace_id"] == "ac-test"
   188→    assert row["entity_id"] == message_value
   189→    assert (row["cpu_ns"], row["adj_cpu_ns"]) == (
   190→        (3, 3) if not plan_bundled_cpus else (3, 1) if plan_bundled_cpus == 2 else (3, 0)
   191→    )
   192→    assert (row["mem_ns"], row["adj_mem_ns"]) == ((4, 4) if not plan_bundled_mem_gib else (4, 3))
   193→
   194→    sandbox_multiplier = row["sandbox_multiplier"]
   195→    nonpreemptible_multiplier = row["nonpreemptible_multiplier"]
   196→    region_multiplier = row["region_multiplier"]
   197→
   198→    sandbox_cpu_ns = row["sandbox_cpu_ns"]
   199→    assert sandbox_cpu_ns == (0 if not sandbox else (sandbox_multiplier - 1) * row["adj_cpu_ns"])
   200→    nonpreemptible_cpu_ns = row["nonpreemptible_cpu_ns"]
   201→    assert nonpreemptible_cpu_ns == (0 if not nonpreemptible else (nonpreemptible_multiplier - 1) * row["adj_cpu_ns"])
   202→    region_cpu_ns = row["region_cpu_ns"]
   203→    assert region_cpu_ns == (row["adj_cpu_ns"] + sandbox_cpu_ns + nonpreemptible_cpu_ns) * (region_multiplier - 1)
   204→
   205→    # Same thing for memory.
   206→    sandbox_mem_ns = row["sandbox_mem_ns"]
   207→    assert sandbox_mem_ns == (0 if not sandbox else (sandbox_multiplier - 1) * row["adj_mem_ns"])
   208→    nonpreemptible_mem_ns = row["nonpreemptible_mem_ns"]
   209→    assert nonpreemptible_mem_ns == (0 if not nonpreemptible else (nonpreemptible_multiplier - 1) * row["adj_mem_ns"])
   210→    region_mem_ns = row["region_mem_ns"]
   211→    assert region_mem_ns == (row["adj_mem_ns"] + sandbox_mem_ns + nonpreemptible_mem_ns) * (region_multiplier - 1)
   212→
   213→    assert (row["gpu_ns"], row["region_gpu_ns"]) == (1, 0.25)
   214→    assert (row["gpu_mem_ns"], row["gpu_temp_ns"]) == (1, 1)
   215→    assert (row["requested_cpu_cores"], row["requested_mem_gib"]) == (0.5, 2)

   229→async def test_query_billable_metrics_by_hour_and_billing_type(state: State, user, clear_metrics_tables):
   230→    """Tests the query_billable_metrics_by_hour_and_billing_type method of TaskMetricsStore."""
   231→    workspace_id = "ac-test"
   232→    environment_id = "en-test"
   233→    start_time = datetime(2023, 1, 1, 0, 0, 0)
   234→    end_time = datetime(2023, 1, 1, 3, 0, 0)
   235→
   236→    # Create test data across multiple hours with different billing types
   237→    resources = api_pb2.Resources(
   238→        milli_cpu=2000,  # 2 CPU cores
   239→        memory_mb=2048,  # 2 GiB
   240→    )
   241→
   242→    ephemeral_workspace_struct = await state.ephemeral.workspaces[user.workspace_id]
   243→    ephemeral_plan_struct = await state.ephemeral.plans[ephemeral_workspace_struct.plan_id]
   244→
   245→    # Add metrics for different billing types across 3 hours
   246→    billing_types = ["cpu", "bill_t4", "bill_a100_40gb"]
   247→    for hour in range(3):
   248→        timestamp = start_time + timedelta(hours=hour)
   249→        for billing_type in billing_types:
   250→            metrics = [
   251→                # Actual usage is less than requested resources
   252→                internal_api_pb2.ResourceMetric(
   253→                    timestamp=timestamp.timestamp(), resource="elapsed_ns", value=3.6e12
   254→                ),  # 1 hour
   255→                internal_api_pb2.ResourceMetric(
   256→                    timestamp=timestamp.timestamp(), resource="cpu_ns", value=1.8e12
   257→                ),  # 0.5 hour
   258→                internal_api_pb2.ResourceMetric(
   259→                    timestamp=timestamp.timestamp(), resource="mem_ns", value=3.6e12
   260→                ),  # 1 hour
   261→                internal_api_pb2.ResourceMetric(
   262→                    timestamp=timestamp.timestamp(), resource="gpu_ns", value=1.8e12
   263→                ),  # 0.5 hour
   264→                internal_api_pb2.ResourceMetric(
   265→                    timestamp=timestamp.timestamp(), resource="function_fee_dollars", value=0.1
   266→                ),
   267→            ]
   268→
   269→            await state.task_metrics_store.track_task_metrics(
   270→                task_id=f"ta-{hour}-{billing_type}",
   271→                workspace_id=workspace_id,
   272→                environment_id=environment_id,
   273→                billing_type=billing_type,
   274→                app_id="ap-test",
   275→                app_type="deployed",
   276→                metrics=metrics,
   277→                resources=resources,
   278→                ephemeral_plan_struct=ephemeral_plan_struct,
   279→                regions=None,
   280→                function_id="fu-test",
   281→                function_name="test_function",
   282→            )
   283→
   284→    await state.mock_clickpipes.task_metrics.join()
   285→
   286→    # Query the metrics
   287→    results = await state.task_metrics_store.query_billable_metrics_by_hour_and_billing_type(
   288→        workspace_id,
   289→        start_time.timestamp(),
   290→        end_time.timestamp(),
   291→    )
   292→
   293→    # Verify results
   294→    assert len(results) == 9  # 3 hours × 3 billing types
   295→
   296→    # Group results by billing type
   297→    results_by_type = {}
   298→    for row in results:
   299→        billing_type = row["billing_type"]
   300→        results_by_type.setdefault(billing_type, []).append(row)
   301→
   302→    # Verify each billing type has correct metrics
   303→    for billing_type in billing_types:
   304→        type_results = results_by_type[billing_type]
   305→        assert len(type_results) == 3  # 3 hours of data
   306→
   307→        for row in type_results:
   308→            # Raw usage metrics
   309→            assert row["cpu_hr"] == 0.5  # 1.8e12 ns = 0.5 hour
   310→            assert row["mem_hr"] == 1.0  # 3.6e12 ns = 1.0 hour
   311→            assert row["gpu_hr"] == 0.5  # 1.8e12 ns = 0.5 hour
   312→
   313→            # Adjusted metrics (based on resource requests)
   314→            # CPU: max(actual, requested) = max(0.5, 2.0) = 2.0
   315→            assert row["adj_cpu_hr"] == 2.0
   316→            # Memory: max(actual, requested) = max(1.0, 2.0) = 2.0
   317→            assert row["adj_mem_hr"] == 2.0
   318→
   319→            # Billing metrics (same as adjusted metrics since no region/sandbox/nonpreemptible multipliers)
   320→            assert row["sandbox_cpu_hr"] == 0
   321→            assert row["sandbox_mem_hr"] == 0
   322→            assert row["nonpreemptible_cpu_hr"] == 0
   323→            assert row["nonpreemptible_mem_hr"] == 0
   324→            assert row["region_cpu_hr"] == 0
   325→            assert row["region_mem_hr"] == 0
   326→            assert row["region_gpu_hr"] == 0
   327→
   328→            assert row["total_function_fee_dollars"] == 0.1
   329→
   330→            # Verify bucket timestamps
   331→            bucket = row["bucket"]
   332→            assert isinstance(bucket, datetime)
   333→            assert bucket.hour in [0, 1, 2]
   334→            assert bucket.minute == 0
   335→            assert bucket.second == 0

   561→def test_region_to_top_level_json_matches_python():
   562→    """Validate that go/taskmetrics/region_to_top_level.json stays in sync with Python.
   563→
   564→    The JSON file is the shared source of truth consumed by the Go sandbox server.
   565→    If this test fails, regenerate the JSON from the repo root:
   566→
   567→        cd server && python -c "
   568→        import json; from modal_server.cloud_provider import REGION_TO_TOP_LEVEL_REGION, TOP_LEVEL_REGIONS
   569→        data = {'top_level_regions': sorted(TOP_LEVEL_REGIONS), 'region_to_top_level': dict(sorted(REGION_TO_TOP_LEVEL_REGION.items()))}
   570→        open('../go/taskmetrics/region_to_top_level.json','w').write(json.dumps(data, indent=2) + chr(10))
   571→        "
   572→    """
   573→    import json
   574→    from pathlib import Path
   575→
   576→    from modal_server.cloud_provider import REGION_TO_TOP_LEVEL_REGION, TOP_LEVEL_REGIONS
   577→
   578→    json_path = Path(__file__).resolve().parents[2] / "go" / "taskmetrics" / "region_to_top_level.json"
   579→    assert json_path.exists(), f"Missing {json_path} — see docstring for generation command"
   580→
   581→    with open(json_path) as f:
   582→        data = json.load(f)
   583→
   584→    json_top_level = set(data["top_level_regions"])
   585→    python_top_level = set(TOP_LEVEL_REGIONS)
   586→    assert json_top_level == python_top_level, (
   587→        f"top_level_regions mismatch.\n"
   588→        f"  In JSON but not Python: {json_top_level - python_top_level}\n"
   589→        f"  In Python but not JSON: {python_top_level - json_top_level}"
   590→    )
   591→
   592→    json_mapping = data["region_to_top_level"]
   593→    python_mapping = REGION_TO_TOP_LEVEL_REGION
   594→    extra_in_json = {k: json_mapping[k] for k in set(json_mapping) - set(python_mapping)}
   595→    extra_in_python = {k: python_mapping[k] for k in set(python_mapping) - set(json_mapping)}
   596→    value_mismatches = {
   597→        k: (json_mapping[k], python_mapping[k])
   598→        for k in set(json_mapping) & set(python_mapping)
   599→        if json_mapping[k] != python_mapping[k]
   600→    }
   601→    assert json_mapping == python_mapping, (
   602→        f"region_to_top_level mismatch.\n"
   603→        f"  In JSON but not Python: {extra_in_json}\n"
   604→        f"  In Python but not JSON: {extra_in_python}\n"
   605→        f"  Value mismatches (json, python): {value_mismatches}"
